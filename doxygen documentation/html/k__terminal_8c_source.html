185 void terminal()
186 {
187  terminal_t term;
188 
189  init_term(&term);
190 
191  ResetScreen();
192  send_header(term.header);
193 
194  uint8_t rx_buf[MSG_MAX_SIZE];
195 
196  ChangeProcessPriority(IDLE_ID, 1);
197 
198  // Alias pointers.
199  // Depending on the src_box value,
200  // rx_buf has two different data structures
201  IO_metadata_t* IO_meta = (IO_metadata_t*)rx_buf;
202  uart_msgdata_t* uart = (uart_msgdata_t*)rx_buf;
203  // The first four bytes in array should be the source box
204  pmbox_t* src_box = (pmbox_t*)&rx_buf;
205 
206  while (1) {
207  recv(term.box, ANY_BOX, rx_buf, MSG_MAX_SIZE);
208 
209  if (*src_box == IO_BOX) {
210  // Process UART input
211  if (uart->c == TERM_ESC) {
212  ResetTerminal(&term);
213  }
214  else if (term.mode == COMMAND_HANDLER || term.capture.en) {
215  ProcessInput(uart->c, &term);
216  }
217  }
218  else if (GetBit(term.active_pid, IO_meta->proc_id) && !term.capture.en) {
219  // Process has IO permissions
220  if (IO_meta->is_send) {
221  UART0_puts((char*)IO_meta->send_data);
222  // Sends data back just so the sender's recv gets the size sent to uart.
223  send(*src_box, term.box, IO_meta->send_data, IO_meta->size);
224  }
225  else {
226  ConfigureInputCapture(&term.capture, IO_meta);
227  }
228  }
229  else {
230  send(*src_box, term.box, "", 0);
231  }
232  }
233 }

318 bool CommandCheck(terminal_t* term)
319 {
320  bool valid_command = false;
321 
322  char* comm = term->buf.data;
323  uint32_t size = term->buf.wr_ptr;
324 
325  char* keyword;
326  char* attr_data;
327 
328  if (!enqueuec_s(&term->buf, '\0', false)) {
329  term->buf.data[term->buf.wr_ptr] = '\0';
330  }
331 
332  int i = 0;
333  // Find the begin of they query entry
334  while (i < size && comm[i] == ' ') i++;
335  keyword = comm + i;
336 
337  // Find the end of the query keyword
338  while (i < size && comm[i] != ' ') i++;
339  comm[i++] = '\0'; // null-terminate the keyword to make decoding easier.
340 
341  // Find the begin of the query set data (if it exists)
342  while (i < size && comm[i] == ' ') i++;
343  attr_data = (i < size) ? (comm + i) : NULL;
344 
345  i = 0;
346  while(i < COMMANDS_SIZE && strcmp(keyword, COMMAND[i]) != 0) i++;
347 
348  if (i < COMMANDS_SIZE) valid_command = CommHandler[i](attr_data, term);
349  if (!valid_command) UART0_puts("?\n> ");
350 
351  return valid_command;
352 }
