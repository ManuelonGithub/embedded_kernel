124 void SVC_handler()
125 {
126  SaveTrapReturn(); // save the trap return address
127 
128  SysTick_Stop();
129 
130  if (TrapSource() == KERNEL) { // Check if the trap was called by the kernel
131  // In which case the kernel needs to save its own context
132  SaveContext();
133  KernelCall_handler(GetCallReg());
134  RestoreContext();
135  }
136  else { // Trap was called by a process
137  SaveProcessContext(); // So the process' context is saved
138  KernelCall_handler(GetCallReg());
139  RestoreProcessContext();
140  }
141 
142  SysTick_Start();
143 
144  RestoreTrapReturn();
145 }
